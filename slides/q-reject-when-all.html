$q - .reject(), .when(), .all()



$q.all({
  aModel: fetchA().then(fetchASiblings),
  bModel: fetchB(),
  err: $q.reject('did not work'), // creates a rejected promise
  msg: $q.when('it worked'), // creates a resolved promise
})
.then(concatData) //receives {aModel:.., bModel:.., err:.., msg: ..}
.then(filterData)
.finally(function () {
  // no arguments
})

$q.all also works with an array
