$q - the deferred API



function getUser (id) {
  var deferred = $q.defer();

  jQuery.get('/api/users/' + id, function onSuccess (user) {
    deferred.resolve(user);
  }, function onError (err) {
    deferred.reject(err);
  });

  return deferred.promise;
}
